    <script>
        // Inisialisasi data siswa dari localStorage
        let siswa = JSON.parse(localStorage.getItem('siswa')) || [];
        
        // Tampilkan data siswa saat halaman dimuat
        document.addEventListener('DOMContentLoaded', function() {
            tampilkanDataSiswa();
        });
        
        // Tangani submit form
        document.getElementById('form-siswa').addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Ambil data dari form
            const nis = document.getElementById('nis').value;
            const nama = document.getElementById('nama').value;
            const tempat_lahir = document.getElementById('tempat_lahir').value;
            const tanggal_lahir = document.getElementById('tanggal_lahir').value;
            const jenis_kelamin = document.getElementById('jenis_kelamin').value;
            const alamat = document.getElementById('alamat').value;
            const jurusan = document.getElementById('jurusan').value;
            
            // Validasi NIS sudah ada
            const cekNis = siswa.find(s => s.nis === nis);
            if (cekNis) {
                tampilkanPesan('NIS sudah terdaftar!', 'error');
                return;
            }
            
            // Buat objek siswa baru
            const siswaBaru = {
                id: Date.now(),
                nis,
                nama,
                tempat_lahir,
                tanggal_lahir,
                jenis_kelamin,
                alamat,
                jurusan
            };
            
            // Tambahkan ke array
            siswa.push(siswaBaru);
            
            // Simpan ke localStorage
            localStorage.setItem('siswa', JSON.stringify(siswa));
            
            // Tampilkan pesan sukses
            tampilkanPesan('Data siswa berhasil disimpan!', 'sukses');
            
            // Reset form
            this.reset();
            
            // Tampilkan data terbaru
            tampilkanDataSiswa();
        });
        
        // Fungsi untuk menampilkan pesan
        function tampilkanPesan(pesan, tipe) {
            const pesanContainer = document.getElementById('pesan-container');
            pesanContainer.innerHTML = `<div class="pesan ${tipe}">${pesan}</div>`;
            
            // Hapus pesan setelah 3 detik
            setTimeout(() => {
                pesanContainer.innerHTML = '';
            }, 3000);
        }
        
        // Fungsi untuk menampilkan data siswa
        function tampilkanDataSiswa() {
            const tabelSiswa = document.getElementById('tabel-siswa');
            
            if (siswa.length === 0) {
                tabelSiswa.innerHTML = '<tr><td colspan="7">Belum ada data siswa</td></tr>';
                return;
            }
            
            let html = '';
            siswa.forEach((s, index) => {
                const tanggalLahirFormatted = new Date(s.tanggal_lahir).toLocaleDateString('id-ID');
                const jenisKelaminText = s.jenis_kelamin === 'L' ? 'Laki-laki' : 'Perempuan';
                
                html += `
                    <tr>
                        <td>${index + 1}</td>
                        <td>${s.nis}</td>
                        <td>${s.nama}</td>
                        <td>${s.tempat_lahir}, ${tanggalLahirFormatted}</td>
                        <td>${jenisKelaminText}</td>
                        <td>${s.jurusan}</td>
                        <td>
                            <button class="btn-hapus" onclick="hapusData(${s.id})">Hapus</button>
                        </td>
                    </tr>
                `;
            });
            
            tabelSiswa.innerHTML = html;
        }
        
        // Fungsi untuk menghapus data
        function hapusData(id) {
            if (confirm('Yakin ingin menghapus data ini?')) {
                // Filter data siswa kecuali yang dihapus
                siswa = siswa.filter(s => s.id !== id);
                
                // Simpan kembali ke localStorage
                localStorage.setItem('siswa', JSON.stringify(siswa));
                
                // Tampilkan pesan sukses
                tampilkanPesan('Data siswa berhasil dihapus!', 'sukses');
                
                // Tampilkan data terbaru
                tampilkanDataSiswa();
            }
        }
    </script>
